## Example

```typescript jsx
import { Search, SearchPrivate } from "@snack-ui/search";

<Search
  value={value}
  onChange={handleSearchChange}
  loading={loading}
  options={options}
  onSubmit={onSubmit}
  size={Search.sizes.S}
  autocomplete
/>

<SearchPrivate
  value={value}
  onChange={handleSearchChange}
  loading={loading}
  onSubmit={onSubmit}
  size={SearchPrivate.sizes.S}
/>
```
