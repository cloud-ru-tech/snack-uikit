# Button

## Installation

`npm i @snack-ui/button`

## Example

```typescript jsx
import {ButtonFilled, ButtonFunction, ButtonSimple} from "@snack-ui/button";
import {Counter} from "@snack-ui/counter";
import {DaySVG} from "@snack-ui/icons";

<ButtonFilled
  label='Navigate'
  onClick={() => navigate()}
  icon={<DaySVG />}
  size={ButtonFilled.sizes.M}
  type={ButtonFilled.types.Critical}
  href='https://sbercloud.ru/'
  target='_blank'
  disabled={true}
  loading={true}
/>

<ButtonSimple
  label='Submit'
  onClick={() => sendForm()}
/>

const CounterProps = { 
  value: 7, 
  appearance: Counter.appearances.Critical,
}

<ButtonFunction
  label='Unread Messages'
  onClick={() => openMessages()}
  counter={CounterProps}
/>
```
