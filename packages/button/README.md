# Button

## Installation

`npm i @snack-uikit/button`

## Example

```typescript jsx
import { ButtonFilled, ButtonFunction, ButtonSimple } from "@snack-uikit/button";
import { Counter } from "@snack-uikit/counter";
import { DaySVG } from "@snack-uikit/icons";

<ButtonFilled
  label='Navigate'
  onClick={() => navigate()}
  icon={<DaySVG />}
  size='m'
  appearance='primary'
  href='https://cloud.ru/'
  target='_blank'
  disabled={true}
  loading={true}
/>

<ButtonSimple
  label='Submit'
  onClick={() => sendForm()}
/>

const CounterProps = { 
  value: 7, 
  appearance: 'red',
}

<ButtonFunction
  label='Unread Messages'
  onClick={() => openMessages()}
  counter={CounterProps}
/>
```
