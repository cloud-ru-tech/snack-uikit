## Example

## Example

```tsx
import { Link } from '@snack-uikit/link';
import { Link as ReactRouterLink } from 'react-router-dom';

<Link
  href='https://cloud.ru/'
  text='Link text'
  size='m'
  appearance='neutral'
  onClick={e => {
    e.stopPropagation();
  }}
/>;

// Polymorphic components

<Link as={ReactRouterLink} text='Link text' size='m' appearance='neutral' to='..' />;
```

|------|------|---------------|-------------|
