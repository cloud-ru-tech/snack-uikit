## Example

```tsx
import { Link } from '@snack-uikit/link';

<Link
  onSurface={Link.onSurfaces.Accent}
  href='https://cloud.ru/'
  text='Link text'
  external={true}
  onColor={Link.onSurfaces.Primary}
/>;
```
